var xpos = 0;
var ypos = 0;


function setup(){
    createCanvas(windowWidth,windowHeight);
    
    angleMode(DEGREES);
    xpos = width/2;
    ypos = height /2;
}


function draw(){
    background(200,1);
    if(frameCount%6 == 0){
        push();
        xpos += random(-10,10);
        ypos += random(-2,5);
        xpos = constrain(xpos,0,width);
        ypos = constrain(ypos,0, height);
        translate(xpos,ypos);


        scale(random(.5,1));

        if(random(0,100)< 10){
             scale(random(3,8));

        }
        rotate(random(0,360));
        fill(random(0,255));
        stroke(100,random(0,100));
        strokeWeight(random(0,20));
        if(random(0,100) < 50){
            //heads
            triangle(0,0,10,25,50,200);
            ellipse(0,0,20,20);
        }else{
            //tails
            fill(random(0,255),random(0,255),random(0,255),random(0,255));
            rect(0,0,40,100);
        }
        
        pop();
    }
}
Generative Art with P5Play Sprites
var xpos = 0;
var ypos = 0;

var firstGroup;
function setup(){

    useQuadTree(false);
    createCanvas(windowWidth,windowHeight);
    
    angleMode(DEGREES);
    xpos = width/2;
    ypos = height /2;

    firstGroup = new Group();

}


function draw(){
    background(200,20);
    if(frameCount%2 == 0){


        xpos+=random(-100,100);
        xpos = wrap(xpos,width);

        ypos+=random(-10,10);
        ypos = wrap(ypos,width);

       var spriteThing = createSprite(xpos,ypos,10,10);
       spriteThing.addSpeed(random(0,10),random(0,360));
       spriteThing.rotation = (random(0,360));
       spriteThing.rotationSpeed = (random(0,10));
       spriteThing.friction = 0.87;
       spriteThing.life = 200;

       firstGroup.add(spriteThing);
    }


    for(var i = 0;i < firstGroup.length;i++){
        
        firstGroup[i].scale+=0.02;
        firstGroup[i].width+=0.03;
        if(random(0,100)< 2){
          firstGroup[i].width+=20; 
        }
        if(random(0,100)< 2){
          firstGroup[i].height-=5; 
        }

        if(random(0,100)< 2){
          firstGroup[i].shapeColor = color(random(0,255),20,20); 
        }

        //kill offscreen particles
        if(firstGroup[i].position.x < 0 || 
        firstGroup[i].position.x > width || 
        firstGroup[i].position.y < 0 || 
        firstGroup[i].position.y > height){
            firstGroup[i].remove();
         }
    }

    firstGroup.bounce(firstGroup);
        
    drawSprites();
}

function wrap(_value,_max){
    
    var wrapper = 0;
    
    if(_value > 0){
        wrapper = _value%_max;
    }
    if(_value <= 0){
        wrapper = _max - (_value%_max);
    }
    
    return wrapper;
    
    
}
